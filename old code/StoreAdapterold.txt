package com.meow.sosapp;

import android.content.Context;
import android.view.LayoutInflater;
import android.view.View;
import android.view.ViewGroup;
import android.widget.BaseAdapter;
import android.widget.TextView;

import java.util.List;

public class StoreAdapter extends BaseAdapter {
    private Context context;
    private List<Store> stores;

    // Constructor and methods to set stores and notifyDataSetChanged()
    public StoreAdapter(Context context, List<Store> stores) {
        this.context = context;
        this.stores = stores;
    }

    @Override
    public int getCount() {
        return stores.size();
    }

    @Override
    public Object getItem(int position) {
        return stores.get(position);
    }

    @Override
    public long getItemId(int position) {
        return position;
    }

    @Override
    public View getView(int position, View convertView, ViewGroup parent) {
        View view = LayoutInflater.from(context).inflate(R.layout.store_item, parent, false);
        TextView storeName = view.findViewById(R.id.store_name);
        TextView storeAddress = view.findViewById(R.id.store_address);
        TextView storeDistance = view.findViewById(R.id.store_distance);
        TextView storePhone = view.findViewById(R.id.store_phone);

        Store store = stores.get(position);
        storeName.setText(store.getName());
        storeAddress.setText(store.getAddress());
        storeDistance.setText(String.format("%.2f km", store.getDistance() / 1000));
        //storePhone.setText(store.getPhoneNumber());

        return view;
    }
    @Override
    public void notifyDataSetChanged() {
        super.notifyDataSetChanged();
    }
}
